function getRequestBody(e) {
  if (e && e.postData && e.postData.contents) {
    try {
      return JSON.parse(e.postData.contents);
    } catch (err) {
      Logger.log('JSON parse failed: %s', err.message);
    }
  }

  var params = {};
  if (e && e.parameter) {
    Object.keys(e.parameter).forEach(function(key) {
      params[key] = e.parameter[key];
    });
  }
  return params;
}

function doPost(e) {
  try {
    var body = getRequestBody(e);
    Logger.log('Received payload: %s', JSON.stringify(body));
    if (!body || Object.keys(body).length === 0) {
      throw new Error("No data received");
    }

    var sheet = SpreadsheetApp
      .openById("1C-XiCMomqdFy_XJsciqd2Q-xNDsdetxGa2VGRBbHXmM")
      .getSheetByName("SWYN EXPERTS DETAILS");

    var folder = DriveApp.getFolderById("1PEFupJKF5Y92tdjfKkIgFwJIPQwUdu4X");

    var formType = body.formType || (e && e.parameter && e.parameter.formType);
    if (formType) {
      Logger.log('Detected formType: %s', formType);
    }

    // ✅ FILE UPLOAD
    if (body.action === "uploadFile") {
      Logger.log('Upload request: type=%s file=%s mime=%s', body.type, body.fileName, body.mimeType);
      if (!body.data || !body.fileName) {
        throw new Error('Missing upload data or fileName');
      }
      var blob = Utilities.newBlob(
        Utilities.base64Decode(body.data),
        body.mimeType || 'application/octet-stream',
        body.fileName
      );

      var file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

      return ContentService
        .createTextOutput(JSON.stringify({
          success: true,
          url: file.getUrl()
        }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ✅ FORM DATA
    if (body) {
      if (!sheet) throw new Error("Sheet not found");

      Logger.log('Appending sheet row: headers=%s values=%s', JSON.stringify(Object.keys(body)), JSON.stringify(Object.values(body)));
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(Object.keys(body));
      }

      sheet.appendRow(Object.values(body));

      return ContentService
        .createTextOutput(JSON.stringify({
          success: true,
          message: "Form submitted"
        }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // fallback
    return ContentService
      .createTextOutput(JSON.stringify({
        success: false,
        error: "Invalid request"
      }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({
        success: false,
        error: err.message
      }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
