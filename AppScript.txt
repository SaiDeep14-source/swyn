function doPost(e) {
  try {
   var sheet = SpreadsheetApp
   .openById("1C-XiCMomqdFy_XJsciqd2Q-xNDsdetxGa2VGRBbHXmM")
    .getSheetByName("SWYN EXPERTS DETAILS");
    
    var folder = DriveApp.getFolderById("1PEFupJKF5Y92tdjfKkIgFwJIPQwUdu4X");
   

    var body = JSON.parse(e.postData.contents);

    // ✅ FILE UPLOAD (base64)
    if (body.action === "uploadFile") {
      var blob = Utilities.newBlob(
        Utilities.base64Decode(body.data),
        body.mimeType,
        body.fileName
      );

      var file = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

      return ContentService
        .createTextOutput(JSON.stringify({ url: file.getUrl() }))
        .setMimeType(ContentService.MimeType.JSON);
    }

    // ✅ FORM DATA
    if (body) {
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(Object.keys(body));
      }
      sheet.appendRow(Object.values(body));

      return ContentService
        .createTextOutput("SUCCESS")
        .setMimeType(ContentService.MimeType.TEXT);
    }

    return ContentService.createTextOutput("INVALID REQUEST");

  } catch (err) {
    return ContentService
      .createTextOutput("ERROR: " + err.message);
  }
}
